\Rightarrow \sum_{i=1}^N (\vec{F_i^a}-\vec{p_i'})\cdot \delta \vec{r_i}+\sum_{i=1}^N \vec{f_i}\cdot \delta \vec{r_i}=0\\
\\
\sum_{i=1}^N (\vec{F_i^a}-\vec{p_i'})\cdot \delta \vec{r_i}=0
\end{align}
